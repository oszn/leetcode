[23. 合并K个升序链表](https://leetcode.cn/problems/merge-k-sorted-lists/)

归并写法，最近写了很多归并，表示右手就行。

```c++
    ListNode* msort(vector<ListNode*>&lists,int start,int end){
        if(start>=end){
            return lists[start];
        }
        int mid=(start+end)/2;
        ListNode*left=  msort(lists,start,mid);
        ListNode*right= msort(lists,mid+1,end);
        ListNode*t=merge(left,right);
        return t;
    }
    ListNode *merge(ListNode*l1,ListNode*l2){
        ListNode*tmp=new ListNode(0);
        ListNode*head=tmp;
        while(l1!=nullptr&&l2!=nullptr){
            if(l1->val>l2->val){
                tmp->next=l2;
                l2=l2->next;
            }else{
                tmp->next=l1;
                l1=l1->next;
            }
            tmp=tmp->next;
        }
        if(l1!=nullptr){
            tmp->next=l1;
        }
        if(l2!=nullptr){
            tmp->next=l2;
        }
        return head->next;
    }
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        int n=lists.size();
        if(n==0){
            return nullptr;
        }
        ListNode*ret=msort(lists,0,n-1);
        return ret;
    }
```

优先队列写法

```c++
 struct node{
        ListNode* ptr;
        friend bool operator  < (node a,node b){
            return a.ptr->val>b.ptr->val;
        };
    };
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        priority_queue<node,vector<node>> litte_queue;
        for(auto c:lists){
            if(c!=nullptr)
            litte_queue.push({c});
        }
        ListNode*head=new ListNode(0);
        ListNode*tail=head;
        while(!litte_queue.empty()){
            node top=litte_queue.top();
            litte_queue.pop();
            tail->next=top.ptr;
            tail=tail->next;
            if(top.ptr->next!=nullptr){
                litte_queue.push({top.ptr->next});
            }
        }
        return head->next;
    }
```
