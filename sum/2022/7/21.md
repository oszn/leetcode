```c++
 bool dfs(TreeNode*rt){
        if(rt==NULL){
            return false;
        }
        bool l=dfs(rt->left);
        bool r=dfs(rt->right);
        
        if(l==false){
            rt->left=NULL;
        }
```

[463. 岛屿的周长](https://leetcode.cn/problems/island-perimeter/)

```c++
    int islandPerimeter(vector<vector<int>>& grid) {
        int m=grid.size();
        if (m==0){
            return 0;
        }
        int x[4]={0,0,1,-1};
        int y[4]={1,-1,0,0};
        int n=grid[0].size();
        int count=0;
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                if(grid[i][j]==1){
                    for(int c=0;c<4;c++){
                        int _i=i+x[c],_j=j+y[c];
                        if(_i>=0&&_i<m&&_j>=0&&_j<n){
                            if(grid[_i][_j]==0){
                                count++;
                            }
                        }else{
                            count++;
                        }
                    }
                }
            }
        }
        return count;
    }
```
